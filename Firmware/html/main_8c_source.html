184 /** ***************************************************************************
185  * @brief Disable the GYRO on the microcontroller board.
186  *
187  * @note MISO of the GYRO is connected to PF8 and CS to PC1.
188  * @n Some times the GYRO goes into an undefined mode at startup
189  * and pulls the MISO low or high thus blocking the analog input on PF8.
190  * @n The simplest solution is to pull the CS of the GYRO low for a short while
191  * which is done with the code below.
192  * @n PF8 is also reconfigured.
193  * @n An other solution would be to remove the GYRO
194  * from the microcontroller board by unsoldering it.
195  *****************************************************************************/
196 static void gyro_disable(void)
197 {
198  __HAL_RCC_GPIOC_CLK_ENABLE(); // Enable Clock for GPIO port C
199  /* Disable PC1 and PF8 first */
200  GPIOC->MODER &= ~GPIO_MODER_MODER1_Msk; // Reset mode for PC1
201  GPIOC->MODER |= 1UL << GPIO_MODER_MODER1_Pos; // Set PC1 as output
202  GPIOC->BSRR |= GPIO_BSRR_BR1; // Set GYRO (CS) to 0 for a short time
203  HAL_Delay(10); // Wait some time
204  GPIOC->MODER |= 3UL << GPIO_MODER_MODER1_Pos; // Analog PC1 = ADC123_IN11
205  __HAL_RCC_GPIOF_CLK_ENABLE(); // Enable Clock for GPIO port F
206  GPIOF->OSPEEDR &= ~GPIO_OSPEEDR_OSPEED8_Msk; // Reset speed of PF8
207  GPIOF->AFR[1] &= ~GPIO_AFRH_AFSEL8_Msk; // Reset alternate function of PF8
208  GPIOF->PUPDR &= ~GPIO_PUPDR_PUPD8_Msk; // Reset pulup/down of PF8
209  HAL_Delay(10); // Wait some time
210  GPIOF->MODER |= 3UL << GPIO_MODER_MODER8_Pos; // Analog mode PF8 = ADC3_IN4
211 }
