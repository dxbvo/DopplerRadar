251 /** ***************************************************************************
252  * @brief Initialize ADC, timer and DMA for data acquisition in the background
253  *
254  * Uses ADC3 and DMA2_Stream1 Channel2
255  * @n The ADC3 trigger is set to TIM2 TRGO event
256  * and the timer starts the ADC directly without CPU intervention.
257  * @ The ADC3 triggers the DMA2_Stream1 to transfer the new data directly
258  * to memory without CPU intervention.
259  * @n The DMA triggers the transfer complete interrupt when all data is ready.
260  * @n The input is ADC3_IN4 = GPIO PF6
261  *****************************************************************************/
262 void ADC3_IN4_DMA_init(void)
263 {
264  MEAS_input_count = 1; // Only 1 input is converted
265  __HAL_RCC_ADC3_CLK_ENABLE(); // Enable Clock for ADC3
266  ADC3->SQR3 |= (4UL << ADC_SQR3_SQ1_Pos); // Input 4 = first conversion
267  ADC3->CR2 |= (1UL << ADC_CR2_EXTEN_Pos); // En. ext. trigger on rising e.
268  ADC3->CR2 |= (6UL << ADC_CR2_EXTSEL_Pos); // Timer 2 TRGO event
269  ADC3->CR2 |= ADC_CR2_DMA; // Enable DMA mode
270  __HAL_RCC_DMA2_CLK_ENABLE(); // Enable Clock for DMA2
271  DMA2_Stream1->CR &= ~DMA_SxCR_EN; // Disable the DMA stream 1
272  while (DMA2_Stream1->CR & DMA_SxCR_EN) { ; } // Wait for DMA to finish
273  DMA2->LIFCR |= DMA_LIFCR_CTCIF1; // Clear transfer complete interrupt fl.
274  DMA2_Stream1->CR |= (2UL << DMA_SxCR_CHSEL_Pos); // Select channel 2
275  DMA2_Stream1->CR |= DMA_SxCR_PL_1; // Priority high
276  DMA2_Stream1->CR |= DMA_SxCR_MSIZE_1; // Memory data size = 32 bit
277  DMA2_Stream1->CR |= DMA_SxCR_PSIZE_1; // Peripheral data size = 32 bit
278  DMA2_Stream1->CR |= DMA_SxCR_MINC; // Increment memory address pointer
279  DMA2_Stream1->CR |= DMA_SxCR_TCIE; // Transfer complete interrupt enable
280  DMA2_Stream1->NDTR = ADC_NUMS; // Number of data items to transfer
281  DMA2_Stream1->PAR = (uint32_t)&ADC3->DR; // Peripheral register address
282  DMA2_Stream1->M0AR = (uint32_t)ADC_samples; // Buffer memory loc. address
283 }

300 /** ***************************************************************************
301  * @brief Initialize ADCs, timer and DMA for simultaneous dual ADC acquisition
302  *
303  * Uses ADC1 and ADC2 in dual mode and DMA2_Stream4 Channel0
304  * @n The ADC1 trigger is set to TIM2 TRGO event
305  * @n ADC1 is the master and simultaneously triggers ADC2.
306  * @n Both converted data from ADC1 and ADC2 are packed into a 32-bit register
307  * in this way: <b> ADC_CDR[31:0] = ADC2_DR[15:0] | ADC1_DR[15:0] </b>
308  * and are transfered with the DMA in one single step.
309  * @n The ADC1 triggers the DMA2_Stream4 to transfer the new data directly
310  * to memory without CPU intervention.
311  * @n The DMA triggers the transfer complete interrupt when all data is ready.
312  * @n The input used with ADC1 is ADC123_IN13 = GPIO PC3
313  * @n The input used with ADC2 is ADC12_IN5 = GPIO PA5
314  *****************************************************************************/
315 void ADC1_IN13_ADC2_IN5_dual_init(void)
316 {
317  MEAS_input_count = 2; // 2 inputs are converted
318  __HAL_RCC_ADC1_CLK_ENABLE(); // Enable Clock for ADC1
319  __HAL_RCC_ADC2_CLK_ENABLE(); // Enable Clock for ADC2
320  ADC->CCR |= ADC_CCR_DMA_1; // Enable DMA mode 2 = dual DMA
321  ADC->CCR |= ADC_CCR_MULTI_1 | ADC_CCR_MULTI_2; // ADC1 and ADC2 simultan.
322  ADC1->CR2 |= (1UL << ADC_CR2_EXTEN_Pos); // En. ext. trigger on rising e.
323  ADC1->CR2 |= (6UL << ADC_CR2_EXTSEL_Pos); // Timer 2 TRGO event
324  ADC1->SQR3 |= (13UL << ADC_SQR3_SQ1_Pos); // Input 13 = first conversion
325  ADC2->SQR3 |= (5UL << ADC_SQR3_SQ1_Pos); // Input 5 = first conversion
326  __HAL_RCC_DMA2_CLK_ENABLE(); // Enable Clock for DMA2
327  DMA2_Stream4->CR &= ~DMA_SxCR_EN; // Disable the DMA stream 4
328  while (DMA2_Stream4->CR & DMA_SxCR_EN) { ; } // Wait for DMA to finish
329  DMA2->HIFCR |= DMA_HIFCR_CTCIF4; // Clear transfer complete interrupt fl.
330  DMA2_Stream4->CR |= (0UL << DMA_SxCR_CHSEL_Pos); // Select channel 0
331  DMA2_Stream4->CR |= DMA_SxCR_PL_1; // Priority high
332  DMA2_Stream4->CR |= DMA_SxCR_MSIZE_1; // Memory data size = 32 bit
333  DMA2_Stream4->CR |= DMA_SxCR_PSIZE_1; // Peripheral data size = 32 bit
334  DMA2_Stream4->CR |= DMA_SxCR_MINC; // Increment memory address pointer
335  DMA2_Stream4->CR |= DMA_SxCR_TCIE; // Transfer complete interrupt enable
336  DMA2_Stream4->NDTR = ADC_NUMS; // Number of data items to transfer
337  DMA2_Stream4->PAR = (uint32_t)&ADC->CDR; // Peripheral register address
338  DMA2_Stream4->M0AR = (uint32_t)ADC_samples; // Buffer memory loc. address
339 }

359  *
360  * Uses ADC2 and DMA2_Stream3 channel1
361  * @n The ADC2 trigger is set to TIM2 TRGO event
362  * @n At each trigger both inputs are converted sequentially
363  * and transfered to memory by the DMA.
364  * @n As each conversion triggers the DMA, the number of transfers is doubled.

544 /** ***************************************************************************
545  * @brief Interrupt handler for DMA2 Stream4
546  *
547  * Here the interrupt handler is used together with ADC1 and ADC2
548  * in dual mode where they sample simultaneously.
549  * @n The samples from both ADCs packed in a 32 bit word have been transfered
550  * to memory by the DMA2 and are ready for unpacking.
551  * @note In dual ADC mode two values are combined (packed) in a single uint32_t
552  * ADC_CDR[31:0] = ADC2_DR[15:0] | ADC1_DR[15:0]
553  * and are therefore extracted before further processing.
554  *****************************************************************************/
555 void DMA2_Stream4_IRQHandler(void)
556 {
557  if (DMA2->HISR & DMA_HISR_TCIF4) { // Stream4 transfer compl. interrupt f.
558  NVIC_DisableIRQ(DMA2_Stream4_IRQn); // Disable DMA interrupt in the NVIC
559  NVIC_ClearPendingIRQ(DMA2_Stream4_IRQn);// Clear pending DMA interrupt
560  DMA2_Stream4->CR &= ~DMA_SxCR_EN; // Disable the DMA
561  while (DMA2_Stream4->CR & DMA_SxCR_EN) { ; } // Wait for DMA to finish
562  DMA2->HIFCR |= DMA_HIFCR_CTCIF4;// Clear transfer complete interrupt fl.
563  TIM2->CR1 &= ~TIM_CR1_CEN; // Disable timer
564  ADC1->CR2 &= ~ADC_CR2_ADON; // Disable ADC1
565  ADC2->CR2 &= ~ADC_CR2_ADON; // Disable ADC2
566  ADC->CCR &= ~ADC_CCR_DMA_1; // Disable DMA mode
567  /* Extract combined samples */
568  for (int32_t i = ADC_NUMS-1; i >= 0; i--){
569  ADC_samples[2*i+1] = (ADC_samples[i] >> 16);
570  ADC_samples[2*i] = (ADC_samples[i] & 0xffff);
571  }
572  ADC_reset();
573  MEAS_data_ready = true;
574  }
575 }
