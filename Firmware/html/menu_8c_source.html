62 /** ***************************************************************************

65  * Each menu entry has two lines.
66  * Text and background colors are applied.
67  * @n These attributes are defined in the variable MENU_draw[].

69 void MENU_draw(void)
70 {
71  BSP_LCD_SetFont(MENU_FONT);
72  uint32_t x, y, m, w, h;
73  y = MENU_Y;
74  m = MENU_MARGIN;
75  w = BSP_LCD_GetXSize()/MENU_ENTRY_COUNT;
76  h = MENU_HEIGHT;
77  for (uint32_t i = 0; i < MENU_ENTRY_COUNT; i++) {
78  x = i*w;
79  BSP_LCD_SetTextColor(MENU_entry[i].back_color);
80  BSP_LCD_FillRect(x+m, y+m, w-2*m, h-2*m);
81  BSP_LCD_SetBackColor(MENU_entry[i].back_color);
82  BSP_LCD_SetTextColor(MENU_entry[i].text_color);
83  BSP_LCD_DisplayStringAt(x+3*m, y+3*m,
84  (uint8_t *)MENU_entry[i].line1, LEFT_MODE);
85  BSP_LCD_DisplayStringAt(x+3*m, y+h/2,
86  (uint8_t *)MENU_entry[i].line2, LEFT_MODE);
87  }

158 /** ***************************************************************************
159  * @brief Check for selection/transition
160  *
161  * If the last transition has been consumed (MENU_NONE == MENU_transition)
162  * and the touchscreen has been touched for a defined period
163  * the variable MENU_transition is set to the touched item.
164  * @note Evalboard revision E (blue PCB) has an inverted y-axis
165  * in the touch controller compared to the display.
166  * Uncomment or comment the <b>\#define EVAL_REV_E</b> in main.h accordingly.
167  *****************************************************************************/
168 void MENU_check_transition(void)
169 {
170  static MENU_item_t item_old = MENU_NONE;
171  static MENU_item_t item_new = MENU_NONE;
172  static TS_StateTypeDef TS_State; // State of the touch controller
173  BSP_TS_GetState(&TS_State); // Get the state
174 
175 
176 // Evalboard revision E (blue) has an inverted y-axis in the touch controller
177 #ifdef EVAL_REV_E
178  TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
179 #endif
180  // Invert x- and y-axis if LCD ist flipped
181 #ifdef FLIPPED_LCD
182  TS_State.X = BSP_LCD_GetXSize() - TS_State.X; // Invert the x-axis
183  TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
184 #endif
185 
186 
187 /*
188  #if (defined(EVAL_REV_E) && !defined(FLIPPED_LCD)) || (!defined(EVAL_REV_E) && defined(FLIPPED_LCD))
189  TS_State.Y = BSP_LCD_GetYSize() - TS_State.Y; // Invert the y-axis
190 #endif
191 #ifdef EVAL_REV_E
192 #endif
193 */
194  if (TS_State.TouchDetected) { // If a touch was detected
195  /* Do only if last transition not pending anymore */
196  if (MENU_NONE == MENU_transition) {
197  item_old = item_new; // Store old item
198  /* If touched within the menu bar? */
199  if ((MENU_Y < TS_State.Y) && (MENU_Y+MENU_HEIGHT > TS_State.Y)) {
200  item_new = TS_State.X // Calculate new item
201  / (BSP_LCD_GetXSize()/MENU_ENTRY_COUNT);
202  if ((0 > item_new) || (MENU_ENTRY_COUNT <= item_new)) {
203  item_new = MENU_NONE; // Out of bounds
204  }
205  if (item_new == item_old) { // 2 times the same menu item
206  item_new = MENU_NONE;
207  MENU_transition = item_old;
208  }
209  }
210  }
211  }
212 }

216 /** ***************************************************************************
217  * @brief Interrupt handler for the touchscreen
218  *
219  * @note BSP_TS_ITConfig(); must be called in the main function
220  * to enable touchscreen interrupt.
221  * @note There are timing issues when interrupt is enabled.
222  * It seems that polling is the better choice with this evaluation board.
223  * @n Call MENU_check_transition() from the while loop in main for polling.
224  *
225  * The touchscreen interrupt is connected to PA15.
226  * @n The interrupt handler for external line 15 to 10 is called.
227  *****************************************************************************/
228 void EXTI15_10_IRQHandler(void)
229 {
230  if (EXTI->PR & EXTI_PR_PR15) { // Check if interrupt on touchscreen
231  EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
232  if (BSP_TS_ITGetStatus()) { // Get interrupt status
233  BSP_TS_ITClear(); // Clear touchscreen controller int.
234  MENU_check_transition();
235  }
236  EXTI->PR |= EXTI_PR_PR15; // Clear pending interrupt on line 15
237  }
238 }
